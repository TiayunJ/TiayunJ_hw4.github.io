Plot examples

Instacart

library(tidyverse)
library(p8105.datasets)

library(plotly)

loading the data and filter the ones under the department ‘alcohol’

data("instacart")

instacart_df = 
  instacart %>% 
  filter(department == "alcohol") %>% view()

plot 1 - bar chart

instacart_df %>% 
  group_by(aisle) %>% 
  summarize(obs = n()) %>% 
  mutate(aisle = fct_reorder(aisle, obs)) %>% 
  plot_ly(x = ~aisle, y = ~obs, color = ~aisle, type = "bar", colors = "viridis")

plot 2 - box plot

instacart_df %>% 
  plot_ly(x = ~aisle, y = ~ order_dow, color = ~aisle, type = "box", colors = "viridis")

plot 3 - line plot

instacart_df %>% 
  group_by(aisle, order_hour_of_day) %>% 
  summarize(obs = n()) %>% 
  plot_ly(x = ~order_hour_of_day, y = ~obs, color = ~aisle, type = "scatter", mode = 'lines+markers', colors = "viridis")
## `summarise()` has grouped output by 'aisle'. You can override using the
## `.groups` argument.
